-- K4 Notes --

-- Git Commands --

-- login --
git config --global user.name 'your name'
git config --global user.email 'you@example.com'

-- branch --
git branch <branch name>
git branch -d <localBranchName>
git push origin --delete <remoteBranchName>

-- init --
git init
git add .
git commit -m 'your commit'

-- remote --
git remote add origin <remote repo url>
git remote -v
git remote rm <remote name>

-- update, clone and upload --
git push origin master
git clone <remote repo url>
git pull origin master

-- C# .NET Commands --
dotnet add package <packageName>
dotnet run
dotnet build -o path/to
dotnet restore

// postgre sql
psql -U postgres
psql --username user --dbname db
grant all privileges on database <db> to <username>;
create user <username> with encrypted password '';
\l --> database list
\d --> table & table structure list
truncate <table name> --> delete all table records
constraint <fk_name> foreign key() references <tablename>(field)
alter table <name> alter column <name> type <datatype>;
alter table <name> rename to <new name>
